function fileread($path)
{
    $handle = fopen($path, "r");
    if(file_exists($path))
    {
        $contents = fread($handle, filesize($path));
        echo $contents;
        fclose($handle);
        return true;
    }
    else
    {
        echo 'File not Found!';
        fclose($handle);
        return false;
    }
